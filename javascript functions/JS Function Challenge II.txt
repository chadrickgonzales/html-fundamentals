function generatePattern() {  
  let result = "";  
  for (let i = 1; i <= 3; i++) {  
    for (let j = 1; j <= i; j++) {  
      result += j + " ";  
    }  
    result += "\n";  
  }  
  return result;  
}

function swapFirstLast(arr) {  
  let first = arr.shift();  
  let last = arr.pop();  
  arr.unshift(last);  
  arr.push(first);  
  return arr;  
}  

let numbers = [10, 20, 30, 40];  
console.log(swapFirstLast(numbers));  
console.log(numbers);  

40 30 20 10  the shift will removes and returns the first element of the array
              the pop will remove and returns the last element of the array
              unshift will add the given value to the begining of the array
              push will add the given value to the end of the array

function filterPassingGrades(grades) {  
  let passing = [];  
  for (let grade of grades) {  
    if (grade >= 70) {  
      passing.push(grade);  
    } else {  
      passing.unshift(grade);  
    }  
  }  
  return passing;  
}  

let scores = [85, 45, 90, 60];

function updateStatus(tasks) {  
  for (let task of tasks) {  
    task.completed = !task.completed;  
  }  
  return tasks;  
}  

let taskList = [  
  { id: 1, completed: false },  
  { id: 2, completed: true }  
];  

console.log(updateStatus(taskList));  
console.log(taskList);

function findValue(arr, target) {  
  for (let i = 0; i < arr.length; i++) {  
    if (arr[i] === target) {  
      return `Found at index ${i}`;  
    }  
  }  
  return "Not found";  
}  

let data = [5, 12, 8, 130, 44];  
console.log(findValue(data, 12));  
console.log(findValue(data, 100));

Not found
